// 全局
let roleList = [];
let artBase64 = "";

// 立绘预览
function previewArt() {
  const file = document.getElementById('artFile').files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    artBase64 = e.target.result;
    const img = document.getElementById('artImg');
    img.src = artBase64;
    img.style.display = "block";
    document.querySelector('#artPreview span').style.display = "none";
  };
  reader.readAsDataURL(file);
}

// 收集数据
function getFormData() {
  return {
    code: document.getElementById('code').value || 'XXX000',
    name: document.getElementById('name').value || '未命名',
    race: document.getElementById('race').value || '未知',
    gender: document.getElementById('gender').value || '未知',
    age: document.getElementById('age').value || '未知',
    height: document.getElementById('height').value || '未知',
    weight: document.getElementById('weight').value || '未知',
    belong: document.getElementById('belong').value || '深空生物工程研究院',
    job: document.getElementById('job').value || '探索队员',
    form: document.getElementById('form').value || '暂无',
    ability: document.getElementById('ability').value || '暂无',
    personality: document.getElementById('personality').value || '暂无',
    story: document.getElementById('story').value || '暂无',
    mark: document.getElementById('mark').value || '暂无',
    art: artBase64
  };
}

// 渲染
function renderList() {
  const box = document.getElementById('rolesList');
  box.innerHTML = roleList.map((r, i) => `
    

      
${r.code}

      
${r.name}

      
    

  `).join('');
}

// 加入列表
function addToList() {
  roleList.push(getFormData());
  renderList();
  alert("已加入批量列表");
}

// 生成单个
function generateSingle() {
  const data = getFormData();
  buildAndShow(data);
}

// 导出一个
function exportOne(index) {
  const data = roleList[index];
  buildAndDownload(data);
}

// 批量导出
function exportBatch() {
  if (roleList.length === 0) {
    alert("请先加入角色到批量列表");
    return;
  }
  roleList.forEach(data => {
    buildAndDownload(data);
  });
  alert(`已批量导出 ${roleList.length} 个角色档案`);
}

// 清空
function clearList() {
  roleList = [];
  renderList();
}

// 生成并显示
function buildAndShow(data) {
  let html = buildHtml(data);
  document.getElementById('codeText').textContent = html;
  document.getElementById('codeOut').style.display = "block";
}

// 生成并下载
function buildAndDownload(data) {
  const html = buildHtml(data);
  const blob = new Blob([html], { type: "text/html" });
  const a = document.createElement("a");
  a.href = URL.createObjectURL(blob);
  a.download = `${data.code}.html`;
  a.click();
}

// 替换模板
function buildHtml(data) {
  return template
    .replace(/{CODE}/g, data.code)
    .replace(/{NAME}/g, data.name)
    .replace(/{RACE}/g, data.race)
    .replace(/{GENDER}/g, data.gender)
    .replace(/{AGE}/g, data.age)
    .replace(/{HEIGHT}/g, data.height)
    .replace(/{WEIGHT}/g, data.weight)
    .replace(/{BELONG}/g, data.belong)
    .replace(/{JOB}/g, data.job)
    .replace(/{FORM}/g, data.form)
    .replace(/{ABILITY}/g, data.ability)
    .replace(/{PERSONALITY}/g, data.personality)
    .replace(/{STORY}/g, data.story)
    .replace(/{MARK}/g, data.mark)
    .replace(/{ART_SRC}/g, data.art || "");
}
